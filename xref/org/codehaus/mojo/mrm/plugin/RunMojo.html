19  import java.net.UnknownHostException;
20  
21  import org.apache.maven.plugin.MojoExecutionException;
22  import org.apache.maven.plugin.MojoFailureException;
23  import org.apache.maven.plugins.annotations.Mojo;
24  import org.apache.maven.plugins.annotations.Parameter;
25  import org.codehaus.plexus.util.FileUtils;
26  import org.codehaus.plexus.util.StringUtils;
27  
28  /**
29   * This goal is used in-situ on a Maven project to allow integration tests based on the Maven Invoker to use a custom
30   * <code>settings.xml</code> and still work behind a proxy.
31   *
32   * @author Stephen Connolly
33   * @description Starts a mock repository manager for manual testing.
34   */
35  @Mojo( name = "run", requiresProject = false, requiresDirectInvocation = true, threadSafe = true )
36  public class RunMojo
37      extends AbstractStartMojo
38  {
39      /**
40       * ServletPath for the settings.xml, so it can be downloaded.
41       */
42      @Parameter( property = "mrm.settingsServletPath", defaultValue = "settings-mrm.xml" )
43      private String settingsServletPath; 
44      
45      /**
46       * {@inheritDoc}
47       */
48      public void doExecute()
49          throws MojoExecutionException, MojoFailureException
50      {
51          if ( !session.getSettings().isInteractiveMode() )
52          {
53              throw new MojoExecutionException(
54                  "Cannot run a mock repository in batch mode (as there is no way to signal shutdown) "
55                      + "use mrm:start instead" );
56          }
57          FileSystemServer mrm = createFileSystemServer( createArtifactStore() );
58          getLog().info( "Starting Mock Repository Manager" );
59          mrm.ensureStarted();
60          String url = mrm.getUrl();
61          try
62          {
63              getLog().info( "Mock Repository Manager " + url + " is started." );
64              if ( StringUtils.isNotEmpty( settingsServletPath ) )
65              {
66                  String downloadUrl;
67                  try
68                  {
69                      downloadUrl = mrm.getRemoteUrl();
70                  }
71                  catch ( UnknownHostException e )
72                  {
73                      downloadUrl = mrm.getUrl();
74                  }
75                  
76                  String settings = FileUtils.filename( settingsServletPath );
77                  
78                  getLog().info( "To share this repository manager, let users download " + downloadUrl + "/" + settingsServletPath );
79                  getLog().info( "Maven should be started as 'mvn --settings " + settings +" [phase|goal]'" );
80              }
81              ConsoleScanner consoleScanner = new ConsoleScanner();
82              consoleScanner.start();
83              getLog().info( "Hit ENTER on the console to stop the Mock Repository Manager and continue the build." );
84              consoleScanner.waitForFinished();
85          }
86          catch ( InterruptedException e )
87          {
88              // ignore
89          }
90          finally
91          {
92              getLog().info( "Stopping Mock Repository Manager " + url );
93              mrm.finish();
94              try
95              {
96                  mrm.waitForFinished();
97                  getLog().info( "Mock Repository Manager " + url + " is stopped." );
98              }
99              catch ( InterruptedException e )
100             {
101                 // ignore
102             }
103         }
104     }
105     
106     /**
107      * {@inheritDoc}
108      */
109     @Override
110     protected String getSettingsServletPath()
111     {
112         return settingsServletPath;
113     }
114 
115 }
