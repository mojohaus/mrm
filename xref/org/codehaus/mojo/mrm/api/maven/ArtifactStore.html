17  package org.codehaus.mojo.mrm.api.maven;
18  
19  import org.apache.maven.archetype.catalog.ArchetypeCatalog;
20  import org.apache.maven.artifact.repository.metadata.Metadata;
21  
22  import java.io.IOException;
23  import java.io.InputStream;
24  import java.io.Serializable;
25  import java.util.Set;
26  
27  /**
28   * An artifact store holds Maven {@link Artifact}s and can provide {@link Metadata} about the artifacts that it holds.
29   *
30   * @since 1.0
31   */
32  public interface ArtifactStore
33      extends Serializable
34  {
35  
36      /**
37       * Returns the set of groupIds that begin with the specified prefix. Some implementations may be lazy caching
38       * implementations, in which case it is permitted to return either the empty set, or only those entries which
39       * have been loaded into the cache, so consumers should not assume that a missing entry implies non-existence.
40       * The only way to be sure that an artifact does not exist is to call the {@link #get(Artifact)} method.
41       * </p>
42       * If there are known to be groupIds: org.codehaus.mojo, org.apache.maven and commons-io then
43       * <pre>
44       * assertEquals(new HashSet&lt;String>(Arrays.asList("commons-io","org")), getGroupIds("")); // Query root level
45       * assertEquals(new HashSet&lt;String>(Arrays.asList("org.codehaus", "org.apache")), getGroupIds("org")); // query with a prefix
46       * assertEquals(new HashSet&lt;String>(Arrays.asList("org.codehaus.mojo")), getGroupIds("org.codehaus"));
47       * </pre>
48       * Note that while the existence of groupId <code>org.codehaus.mojo</code> implies that there must be groupIds
49       * <code>org</code> and <code>org.codehaus</code> there is no requirement that an implementation should
50       * report these inferred parent groupIds, it is just strongly encouraged.
51       * </p>
52       * Note that where an implementation cannot differentiate between artifactIds, child groupIds and perhaps even
53       * versions, it is permitted to return all three.
54       *
55       * @param parentGroupId The prefix to query or the empty string to query the root, cannot be <code>null</code>.
56       * @return A set (with all elements of type {@link String}) of groupIds that are known to have
57       *         {@code parentGroupId} as their prefix. All returned elements must start with {@code parentGroupId} and must have
58       *         one and only one additional segment.
59       * @since 1.0
60       */
61      Set<String> getGroupIds( String parentGroupId );
62  
63      /**
64       * Returns the set of artifactIds that belong in the specified groupId. Some implementations may be lazy caching
65       * implementations, in which case it is permitted to return either the empty set, or only those entries which
66       * have been loaded into the cache, so consumers should not assume that a missing entry implies non-existence.
67       * The only way to be sure that an artifact does not exist is to call the {@link #get(Artifact)} method.
68       * </p>
69       * Note that where an implementation cannot differentiate between artifactIds, child groupIds and perhaps even
70       * versions, it is permitted to return all three.
71       *
72       * @param groupId The groupId to query cannot be empty or <code>null</code>.
73       * @return A set (with all elements of type {@link String}) of artifactIds that are known to belong to
74       *         {@code groupId}.
75       * @since 1.0
76       */
77      Set<String> getArtifactIds( String groupId );
78  
79      /**
80       * Returns the set of versions of the specified groupId:artifactId. Some implementations may be lazy caching
81       * implementations, in which case it is permitted to return either the empty set, or only those entries which
82       * have been loaded into the cache, so consumers should not assume that a missing entry implies non-existence.
83       * The only way to be sure that an artifact does not exist is to call the {@link #get(Artifact)} method.
84       * </p>
85       * Note that where an implementation cannot differentiate between artifactIds, child groupIds and perhaps even
86       * versions, it is permitted to return all three.
87       *
88       * @param groupId    The groupId to query cannot be empty or <code>null</code>.
89       * @param artifactId The artifactId to query cannot be empty or <code>null</code>.
90       * @return A set (with all elements of type {@link String}) of versions that are known to exist for
91       *         {@code groupId:artifactId}.
92       * @since 1.0
93       */
94      Set<String> getVersions( String groupId, String artifactId );
95  
96      /**
97       * Returns the set of artifacts at the specified groupId:artifactId:version. Some implementations may be lazy caching
98       * implementations, in which case it is permitted to return either the empty set, or only those entries which
99       * have been loaded into the cache, so consumers should not assume that a missing entry implies non-existence.
100      * The only way to be sure that an artifact does not exist is to call the {@link #get(Artifact)} method.
101      *
102      * @param groupId    The groupId to query cannot be empty or <code>null</code>.
103      * @param artifactId The artifactId to query cannot be empty or <code>null</code>.
104      * @param version    The version to query cannot be empty or <code>null</code>.
105      * @return A set (with all elements of type {@link Artifact} of artifacts that are known to exist of the
106      *         specified {@code groupId:artifactId:version}.
107      * @since 1.0
108      */
109     Set<Artifact> getArtifacts( String groupId, String artifactId, String version );
110 
111     /**
112      * Returns the time that the specified artifact was last modified.
113      *
114      * @param artifact the artifact.
115      * @return A <code>long</code> value representing the time the file was
116      *         last modified, measured in milliseconds since the epoch
117      *         (00:00:00 GMT, January 1, 1970), or <code>0L</code> if the
118      *         artifact might not exist (where the artifact is known to not exist an
119      *         {@link ArtifactNotFoundException} must be thrown.
120      * @throws IOException               if an I/O error occurs.
121      * @throws ArtifactNotFoundException if the artifact definitely does not exist.
122      * @since 1.0
123      */
124     long getLastModified( Artifact artifact )
125         throws IOException, ArtifactNotFoundException;
126 
127     /**
128      * Returns the size in bytes of the specified artifact.
129      *
130      * @param artifact the artifact.
131      * @return the length of the artifact in bytes or <code>-1L</code> if the length cannot be determined.
132      * @throws IOException               if an I/O error occurs.
133      * @throws ArtifactNotFoundException if the artifact definitely does not exist.
134      * @since 1.0
135      */
136     long getSize( Artifact artifact )
137         throws IOException, ArtifactNotFoundException;
138 
139     /**
140      * Retrieves the the artifact as an {@link InputStream}. The caller is responsible for closing the stream.
141      *
142      * @param artifact the artifact.
143      * @return the contents of the artifact (caller is responsible for closing).
144      * @throws IOException               if the artifact could not be retrieved.
145      * @throws ArtifactNotFoundException if the artifact does not exist.
146      * @since 1.0
147      */
148     InputStream get( Artifact artifact )
149         throws IOException, ArtifactNotFoundException;
150 
151     /**
152      * Create/update the specified artifact. This is an optional method for implementers.
153      *
154      * @param artifact the artifact to create/update.
155      * @param content  the stream of contents (implementer is responsible for closing).
156      * @throws IOException                   if the content could not be read/written.
157      * @throws UnsupportedOperationException if the implementation is a read-only implementation.
158      * @since 1.0
159      */
160     void set( Artifact artifact, InputStream content )
161         throws IOException;
162 
163     /**
164      * Returns the specified metadata.
165      *
166      * @param path of the metadata (should not include the <code>maven-metadata.xml</code>.
167      * @return the metadata, never <code>null</code>.
168      * @throws IOException               if an I/O error occurs.
169      * @throws MetadataNotFoundException if the metadata does not exist.
170      * @since 1.0
171      */
172     Metadata getMetadata( String path )
173         throws IOException, MetadataNotFoundException;
174 
175     /**
176      * Create/update the specified metadata.
177      * 
178      * @param path of the metadata (should not include the <code>maven-metadata.xml</code>.
179      * @param content the metadata, never <code>null</code>.
180      * @throws IOException if an I/O error occurs.
181      * @since 1.1.0
182      */
183     void setMetadata( String path, Metadata content ) throws IOException;
184 
185     /**
186      * Returns the time that the specified metadata was last modified.
187      *
188      * @param path of the metadata (should not include the <code>maven-metadata.xml</code>.
189      * @return A <code>long</code> value representing the time the file was
190      *         last modified, measured in milliseconds since the epoch
191      *         (00:00:00 GMT, January 1, 1970), or <code>0L</code> if the
192      *         metadata might not exist (where the metadtat is known to not exist a
193      *         {@link MetadataNotFoundException} must be thrown.
194      * @throws IOException               if an I/O error occurs.
195      * @throws MetadataNotFoundException if the metadata definitely does not exist.
196      * @since 1.0
197      */
198     long getMetadataLastModified( String path )
199         throws IOException, MetadataNotFoundException;
200     
201     /**
202      * 
203      * @return 
204      * @throws IOException                       if an I/O error occurs.
205      * @throws ArchetypeCatalogNotFoundException if the archetypeCatalog does not exist.
206      * @since 1.0
207      */
208     ArchetypeCatalog getArchetypeCatalog() throws IOException, ArchetypeCatalogNotFoundException;
209     
210     /**
211      * 
212      * @return
213      * @throws IOException                       if an I/O error occurs.
214      * @throws ArchetypeCatalogNotFoundException if the archetypeCatalog does not exist.
215      * @since 1.0
216      */
217     long getArchetypeCatalogLastModified() throws IOException, ArchetypeCatalogNotFoundException;
218 
219     /**
220      * 
221      * @param content
222      * @throws IOException                       if an I/O error occurs.
223      * @since 1.0
224      */
225     void setArchetypeCatalog( InputStream content ) throws IOException;
226 
227 }
