50      /**
51       * Returns an input stream for the specified content.  If the content is a byte array, the input stream will be a
52       * {@link java.io.ByteArrayInputStream} if the content is a File, the input stream will be a
53       * {@link java.io.FileInputStream} otherwise the content will be converted to a String and then into its UTF-8
54       * representation and a {@link java.io.ByteArrayInputStream} returned.
55       *
56       * @param content The content.
57       * @return an input stream of the content.
58       * @throws java.io.IOException if things go wrong.
59       * @since 1.0
60       */
61      public static InputStream asInputStream( Object content )
62          throws IOException
63      {
64          if ( content instanceof byte[] )
65          {
66              return new ByteArrayInputStream( (byte[]) content );
67          }
68          else if ( content instanceof File )
69          {
70              return new FileInputStream( (File) content );
71          }
72          else
73          {
74              return new ByteArrayInputStream( content.toString().getBytes(StandardCharsets.UTF_8) );
75          }
76      }

78      /**
79       * Creates an empty jar file.
80       *
81       * @return the empty jar file as a byte array.
82       * @throws IOException if things go wrong.
83       * @since 1.0
84       */
85      public static byte[] newEmptyJarContent()
86          throws IOException
87      {
88          final Manifest manifest = new Manifest();
89          manifest.getMainAttributes().putValue( "Manifest-Version", "1.0" );
90          manifest.getMainAttributes().putValue( "Archiver-Version", "1.0" );
91          manifest.getMainAttributes().putValue( "Created-By", "Mock Repository Maven Plugin" );
92  
93          ByteArrayOutputStream bos = new ByteArrayOutputStream();
94          new JarOutputStream( bos, manifest ).close();
95          return bos.toByteArray();
96      }

98      /**
99       * Creates an empty maven plugin jar file.
100      *
101      * @param groupId    the group id of the plugin.
102      * @param artifactId the artifact id of the plugin.
103      * @param version    the version of the plugin.
104      * @return the empty jar file as a byte array.
105      * @throws IOException if things go wrong.
106      * @since 1.0
107      */
108     public static byte[] newEmptyMavenPluginJarContent( String groupId, String artifactId, String version )
109         throws IOException
110     {
111         final Manifest manifest = new Manifest();
112         manifest.getMainAttributes().putValue( "Manifest-Version", "1.0" );
113         manifest.getMainAttributes().putValue( "Archiver-Version", "1.0" );
114         manifest.getMainAttributes().putValue( "Created-By", "Mock Repository Maven Plugin" );
115 
116         ByteArrayOutputStream bos = new ByteArrayOutputStream();
117         try ( JarOutputStream jos = new JarOutputStream( bos, manifest ) )
118         {
119             JarEntry entry = new JarEntry( "META-INF/maven/plugin.xml" );
120             jos.putNextEntry( entry );
121             jos.write(
122                 ( "<plugin><groupId>" + groupId + "</groupId><artifactId>" + artifactId + "</artifactId><version>"
123                     + version
124                     + "</version></plugin>" ).getBytes() );
125             jos.closeEntry();
126         }
127         return bos.toByteArray();
128     }

130     /**
131      * Converts a GAV coordinate into the base file path and name for all artifacts at that coordinate.
132      *
133      * @param groupId    the group id.
134      * @param artifactId the artifact id.
135      * @param version    the version.
136      * @return the base filepath for artifacts at the specified coordinates.
137      * @since 1.0
138      */
139     public static String getGAVPathName( String groupId, String artifactId, String version )
140     {
141         return getGAVPath( groupId, artifactId, version ) + '/' + artifactId + '-' + version;
142     }
143 
144     /**
145      * Converts a GAV coordinate into the repository path for the directory containing all artifacts at that GAV.
146      *
147      * @param groupId    the group id.
148      * @param artifactId the artifact id (may be <code>null</code> to just get the path of the groupId)
149      * @param version    the version (may be <code>null</code> to just get the path of the groupId:artifactId)
150      * @return the path.
151      * @since 1.0
152      */
153     public static String getGAVPath( String groupId, String artifactId, String version )
154     {
155         return groupId.replace( '.', '/' ) + ( artifactId != null ? ( '/' + artifactId + ( version != null ? ( '/'
156             + version ) : "" ) ) : "" );
157     }

210     /**
211      * Take a path and encode it for use as an URL parameter.
212      *
213      * @param path the path.
214      * @return the path encoded for use as an URL parameter.
215      * @throws UnsupportedEncodingException if the path cannot be encoded.
216      * @since 1.0
217      */
218     public static String urlEncodePath( String path )
219         throws UnsupportedEncodingException
220     {
221         StringBuilder buf = new StringBuilder( path.length() + 64 );
222         int last = 0;
223         for ( int i = path.indexOf( '/' ); i != -1; i = path.indexOf( '/', last ) )
224         {
225             buf.append( urlEncodePathSegment( path.substring( last, i ) ) );
226             buf.append(path, i, Math.min( path.length(), i + 1 ));
227             last = i + 1;
228         }
229         buf.append( path.substring( last ) );
230         return buf.toString();
231     }

233     /**
234      * Take a path segment and encode it for use as an URL parameter.
235      *
236      * @param pathSegment the path segment.
237      * @return the path segment encoded for use as an URL parameter.
238      * @throws UnsupportedEncodingException if the path cannot be encoded.
239      * @since 1.0
240      */
241     public static String urlEncodePathSegment( String pathSegment )
242         throws UnsupportedEncodingException
243     {
244         StringBuilder buf = new StringBuilder( pathSegment.length() + 64 );
245         byte[] chars = pathSegment.getBytes(StandardCharsets.UTF_8);
246         for (byte aChar : chars) {
247             switch (aChar) {
248                 case '$':
249                 case '-':
250                 case '_':
251                 case '.':
252                 case '!':
253                 case '*':
254                 case '\'':
255                 case '(':
256                 case ')':
257                 case ',':
258                 case '0':
259                 case '1':
260                 case '2':
261                 case '3':
262                 case '4':
263                 case '5':
264                 case '6':
265                 case '7':
266                 case '8':
267                 case '9':
268                 case 'A':
269                 case 'B':
270                 case 'C':
271                 case 'D':
272                 case 'E':
273                 case 'F':
274                 case 'G':
275                 case 'H':
276                 case 'I':
277                 case 'J':
278                 case 'K':
279                 case 'L':
280                 case 'M':
281                 case 'N':
282                 case 'O':
283                 case 'P':
284                 case 'Q':
285                 case 'R':
286                 case 'S':
287                 case 'T':
288                 case 'U':
289                 case 'V':
290                 case 'W':
291                 case 'X':
292                 case 'Y':
293                 case 'Z':
294                 case 'a':
295                 case 'b':
296                 case 'c':
297                 case 'd':
298                 case 'e':
299                 case 'f':
300                 case 'g':
301                 case 'h':
302                 case 'i':
303                 case 'j':
304                 case 'k':
305                 case 'l':
306                 case 'm':
307                 case 'n':
308                 case 'o':
309                 case 'p':
310                 case 'q':
311                 case 'r':
312                 case 's':
313                 case 't':
314                 case 'u':
315                 case 'v':
316                 case 'w':
317                 case 'x':
318                 case 'y':
319                 case 'z':
320                     buf.append((char) aChar);
321                     break;
322                 case ' ':
323                     buf.append('+');
324                     break;
325                 default:
326                     buf.append('%');
327                     if ((aChar & 0xf0) == 0) {
328                         buf.append('0');
329                     }
330                     buf.append(Integer.toHexString(aChar & 0xff));
331                     break;
332             }
333         }
334         return buf.toString();
335     }
