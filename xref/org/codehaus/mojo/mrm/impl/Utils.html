49      /**
50       * Returns an input stream for the specified content.  If the content is a byte array, the input stream will be a
51       * {@link java.io.ByteArrayInputStream} if the content is a File, the input stream will be a
52       * {@link java.io.FileInputStream} otherwise the content will be converted to a String and then into its UTF-8
53       * representation and a {@link java.io.ByteArrayInputStream} returned.
54       *
55       * @param content The content.
56       * @return an input stream of the content.
57       * @throws java.io.IOException if things go wrong.
58       * @since 1.0
59       */
60      public static InputStream asInputStream( Object content )
61          throws IOException
62      {
63          if ( content instanceof byte[] )
64          {
65              return new ByteArrayInputStream( (byte[]) content );
66          }
67          else if ( content instanceof File )
68          {
69              return new FileInputStream( (File) content );
70          }
71          else
72          {
73              return new ByteArrayInputStream( content.toString().getBytes( "UTF-8" ) );
74          }
75      }

77      /**
78       * Creates an empty jar file.
79       *
80       * @return the empty jar file as a byte array.
81       * @throws IOException if things go wrong.
82       * @since 1.0
83       */
84      public static byte[] newEmptyJarContent()
85          throws IOException
86      {
87          byte[] emptyJar;
88          ByteArrayOutputStream bos = new ByteArrayOutputStream();
89          final Manifest manifest = new Manifest();
90          manifest.getMainAttributes().putValue( "Manifest-Version", "1.0" );
91          manifest.getMainAttributes().putValue( "Archiver-Version", "1.0" );
92          manifest.getMainAttributes().putValue( "Created-By", "Mock Repository Maven Plugin" );
93          JarOutputStream jos = new JarOutputStream( bos, manifest );
94          jos.close();
95          bos.close();
96          emptyJar = bos.toByteArray();
97          return emptyJar;
98      }

100     /**
101      * Creates an empty maven plugin jar file.
102      *
103      * @param groupId    the group id of the plugin.
104      * @param artifactId the artifact id of the plugin.
105      * @param version    the version of the plugin.
106      * @return the empty jar file as a byte array.
107      * @throws IOException if things go wrong.
108      * @since 1.0
109      */
110     public static byte[] newEmptyMavenPluginJarContent( String groupId, String artifactId, String version )
111         throws IOException
112     {
113         byte[] emptyJar;
114         ByteArrayOutputStream bos = new ByteArrayOutputStream();
115         final Manifest manifest = new Manifest();
116         manifest.getMainAttributes().putValue( "Manifest-Version", "1.0" );
117         manifest.getMainAttributes().putValue( "Archiver-Version", "1.0" );
118         manifest.getMainAttributes().putValue( "Created-By", "Mock Repository Maven Plugin" );
119         JarOutputStream jos = new JarOutputStream( bos, manifest );
120         JarEntry entry = new JarEntry( "META-INF/maven/plugin.xml" );
121         jos.putNextEntry( entry );
122         jos.write(
123             ( "<plugin><groupId>" + groupId + "</groupId><artifactId>" + artifactId + "</artifactId><version>" + version
124                 + "</version></plugin>" ).getBytes() );
125         jos.closeEntry();
126         jos.close();
127         bos.close();
128         emptyJar = bos.toByteArray();
129         return emptyJar;
130     }

132     /**
133      * Converts a GAV coordinate into the base file path and name for all artifacts at that coordinate.
134      *
135      * @param groupId    the group id.
136      * @param artifactId the artifact id.
137      * @param version    the version.
138      * @return the base filepath for artifacts at the specified coordinates.
139      * @since 1.0
140      */
141     public static String getGAVPathName( String groupId, String artifactId, String version )
142     {
143         return getGAVPath( groupId, artifactId, version ) + '/' + artifactId + '-' + version;
144     }
145 
146     /**
147      * Converts a GAV coordinate into the repository path for the directory containing all artifacts at that GAV.
148      *
149      * @param groupId    the group id.
150      * @param artifactId the artifact id (may be <code>null</code> to just get the path of the groupId)
151      * @param version    the version (may be <code>null</code> to just get the path of the groupId:artifactId)
152      * @return the path.
153      * @since 1.0
154      */
155     public static String getGAVPath( String groupId, String artifactId, String version )
156     {
157         return groupId.replace( '.', '/' ) + ( artifactId != null ? ( '/' + artifactId + ( version != null ? ( '/'
158             + version ) : "" ) ) : "" );
159     }

212     /**
213      * Take a path and encode it for use as an URL parameter.
214      *
215      * @param path the path.
216      * @return the path encoded for use as an URL parameter.
217      * @throws UnsupportedEncodingException if the path cannot be encoded.
218      * @since 1.0
219      */
220     public static String urlEncodePath( String path )
221         throws UnsupportedEncodingException
222     {
223         StringBuffer buf = new StringBuffer( path.length() + 64 );
224         int last = 0;
225         for ( int i = path.indexOf( '/' ); i != -1; i = path.indexOf( '/', last ) )
226         {
227             buf.append( urlEncodePathSegment( path.substring( last, i ) ) );
228             buf.append( path.substring( i, Math.min( path.length(), i + 1 ) ) );
229             last = i + 1;
230         }
231         buf.append( path.substring( last ) );
232         return buf.toString();
233     }

235     /**
236      * Take a path segment and encode it for use as an URL parameter.
237      *
238      * @param pathSegment the path segment.
239      * @return the path segment encoded for use as an URL parameter.
240      * @throws UnsupportedEncodingException if the path cannot be encoded.
241      * @since 1.0
242      */
243     public static String urlEncodePathSegment( String pathSegment )
244         throws UnsupportedEncodingException
245     {
246         StringBuffer buf = new StringBuffer( pathSegment.length() + 64 );
247         byte[] chars = pathSegment.getBytes( "UTF-8" );
248         for ( int i = 0; i < chars.length; i++ )
249         {
250             switch ( chars[i] )
251             {
252                 case '$':
253                 case '-':
254                 case '_':
255                 case '.':
256                 case '!':
257                 case '*':
258                 case '\'':
259                 case '(':
260                 case ')':
261                 case ',':
262                 case '0':
263                 case '1':
264                 case '2':
265                 case '3':
266                 case '4':
267                 case '5':
268                 case '6':
269                 case '7':
270                 case '8':
271                 case '9':
272                 case 'A':
273                 case 'B':
274                 case 'C':
275                 case 'D':
276                 case 'E':
277                 case 'F':
278                 case 'G':
279                 case 'H':
280                 case 'I':
281                 case 'J':
282                 case 'K':
283                 case 'L':
284                 case 'M':
285                 case 'N':
286                 case 'O':
287                 case 'P':
288                 case 'Q':
289                 case 'R':
290                 case 'S':
291                 case 'T':
292                 case 'U':
293                 case 'V':
294                 case 'W':
295                 case 'X':
296                 case 'Y':
297                 case 'Z':
298                 case 'a':
299                 case 'b':
300                 case 'c':
301                 case 'd':
302                 case 'e':
303                 case 'f':
304                 case 'g':
305                 case 'h':
306                 case 'i':
307                 case 'j':
308                 case 'k':
309                 case 'l':
310                 case 'm':
311                 case 'n':
312                 case 'o':
313                 case 'p':
314                 case 'q':
315                 case 'r':
316                 case 's':
317                 case 't':
318                 case 'u':
319                 case 'v':
320                 case 'w':
321                 case 'x':
322                 case 'y':
323                 case 'z':
324                     buf.append( (char) chars[i] );
325                     break;
326                 case ' ':
327                     buf.append( '+' );
328                     break;
329                 default:
330                     buf.append( '%' );
331                     if ( ( chars[i] & 0xf0 ) == 0 )
332                     {
333                         buf.append( '0' );
334                     }
335                     buf.append( Integer.toHexString( chars[i] & 0xff ) );
336                     break;
337             }
338         }
339         return buf.toString();
340     }
